---
title: useAddPasskey
description: Register a new passkey for the current user.
---

Prompts the WebAuthn ceremony then refetches the passkey list on success.

## Usage

```tsx
import { useAuth } from "@better-auth-ui/react"
import {
  type PasskeyAuthClient,
  useAddPasskey
} from "@better-auth-ui/react/plugins/passkey"

const { authClient } = useAuth<PasskeyAuthClient>()
const { mutate: addPasskey, isPending } = useAddPasskey(authClient)

addPasskey({ name: "MacBook Pro" })
```

`useAddPasskey` requires the Better Auth `passkey` plugin. Type `useAuth` with `PasskeyAuthClient` so TypeScript picks up the plugin-extended method signature.

## Options factory

```tsx
import { addPasskeyOptions } from "@better-auth-ui/react/plugins/passkey"
import { useMutation } from "@tanstack/react-query"

const { mutate } = useMutation(addPasskeyOptions(authClient))
```
