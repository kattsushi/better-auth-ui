---
title: useSignInMagicLink
description: Send a magic-link sign-in email.
---

The session is established when the user clicks the emailed link, so no session-cache side effects run here.

## Usage

```tsx
import { useAuth } from "@better-auth-ui/react"
import {
  type MagicLinkAuthClient,
  useSignInMagicLink
} from "@better-auth-ui/react/plugins/magic-link"

const { authClient } = useAuth<MagicLinkAuthClient>()
const { mutate: signInMagicLink } = useSignInMagicLink(authClient)

signInMagicLink({
  email: "alice@example.com",
  callbackURL: "/dashboard"
})
```

`useSignInMagicLink` requires the Better Auth `magicLink` plugin. Type `useAuth` with `MagicLinkAuthClient` so TypeScript picks up the plugin-extended method signature.

## Options factory

```tsx
import { signInMagicLinkOptions } from "@better-auth-ui/react/plugins/magic-link"
import { useMutation } from "@tanstack/react-query"

const { mutate } = useMutation(signInMagicLinkOptions(authClient))
```
