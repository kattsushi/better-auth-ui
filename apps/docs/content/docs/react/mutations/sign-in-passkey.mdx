---
title: useSignInPasskey
description: Passkey (WebAuthn) sign-in mutation.
---

Resets the session query on success so the new session is refetched.

## Usage

```tsx
import { useAuth } from "@better-auth-ui/react"
import type { PasskeyAuthClient } from "@better-auth-ui/core/plugins/passkey"
import { useSignInPasskey } from "@better-auth-ui/react/plugins/passkey"

const { authClient } = useAuth<PasskeyAuthClient>()
const { mutate: signInPasskey } = useSignInPasskey(authClient)

signInPasskey()
```

`useSignInPasskey` requires the Better Auth `passkey` plugin. Type `useAuth` with `PasskeyAuthClient` so TypeScript picks up the plugin-extended method signature.

## Options factory

```tsx
import { signInPasskeyOptions } from "@better-auth-ui/core/plugins/passkey"
import { useMutation } from "@tanstack/react-query"

const { mutate } = useMutation(signInPasskeyOptions(authClient))
```
