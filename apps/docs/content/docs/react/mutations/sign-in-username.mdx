---
title: useSignInUsername
description: Username/password sign-in mutation.
---

Resets the session query on success so the new session is refetched.

## Usage

```tsx
import { useAuth } from "@better-auth-ui/react"
import type { UsernameAuthClient } from "@better-auth-ui/core/plugins/username"
import { useSignInUsername } from "@better-auth-ui/react/plugins/username"

const { authClient } = useAuth<UsernameAuthClient>()
const { mutate: signInUsername } = useSignInUsername(authClient)

signInUsername({
  username: "alice",
  password: "hunter2"
})
```

`useSignInUsername` requires the Better Auth `username` plugin. Type `useAuth` with `UsernameAuthClient` so TypeScript picks up the plugin-extended method signature.

## Options factory

```tsx
import { signInUsernameOptions } from "@better-auth-ui/core/plugins/username"
import { useMutation } from "@tanstack/react-query"

const { mutate } = useMutation(signInUsernameOptions(authClient))
```
