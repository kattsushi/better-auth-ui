---
title: useUpdateOrganization
description: Update an organization.
---

Organization mutations require `organizationClient()` on the Better Auth client.

## Usage

```tsx
import { useAuth } from "@better-auth-ui/react"
import { useUpdateOrganization } from "@better-auth-ui/react/plugins/organization"

const { authClient } = useAuth()
const mutation = useUpdateOrganization(authClient)

mutation.mutate(/* params */)
```

## Options factory

```tsx
import { updateOrganizationOptions } from "@better-auth-ui/core/plugins/organization"
import { useMutation } from "@tanstack/react-query"

const mutation = useMutation(updateOrganizationOptions(authClient))
```
