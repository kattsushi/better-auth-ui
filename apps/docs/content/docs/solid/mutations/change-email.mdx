## Usage

```tsx
import { changeEmailOptions } from "@better-auth-ui/solid"
import { createMutation } from "@tanstack/solid-query"

const changeEmail = createMutation(() => changeEmailOptions(authClient))

changeEmail.mutate({ newEmail: "alice@example.com" })
```
