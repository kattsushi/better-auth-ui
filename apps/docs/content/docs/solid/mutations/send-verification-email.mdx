## Usage

```tsx
import { sendVerificationEmailOptions } from "@better-auth-ui/solid"
import { createMutation } from "@tanstack/solid-query"

const sendVerificationEmail = createMutation(() =>
  sendVerificationEmailOptions(authClient)
)

sendVerificationEmail.mutate({ email: "alice@example.com" })
```
