## Usage

```tsx
import { signInEmailOptions } from "@better-auth-ui/solid"
import { createMutation } from "@tanstack/solid-query"

const signIn = createMutation(() => signInEmailOptions(authClient))

signIn.mutate({ email: "alice@example.com", password: "hunter2" })
```

## Cache

Invalidate `authQueryKeys.session` on success when your screen needs the new session immediately.
